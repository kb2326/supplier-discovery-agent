⚡ Real world — Neo4j Cypher query
// Find all suppliers of Siemens + their risk
MATCH (s:Supplier)
  -[:SUPPLIES_TO]─►
  (c:Company { name: "Siemens" })
OPTIONAL MATCH
  (p:ParentCo)-[:OWNS]─►(s)
OPTIONAL MATCH
  (s)-[:CERTIFIED_BY]─►(cb:CertBody)
RETURN
  s.name, s.country,
  p.name       AS parent,
  p.ofac_hit   AS sanctions_risk,
  cb.name      AS cert_body
ORDER BY s.financial_health DESC

// Real data: GLEIF LEI for ownership chain
// OFAC API for sanctions screening
